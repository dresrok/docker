# Loop
docker build -t loop .
docker run -d --name looper loop
docker ps

docker stop looper
# Show last process
docker ps -l
docker rm looper

docker run -d --name looper loop
# kill with SIGTERM
docker kill looper
docker rm looper

docker run -d --name looper loop
# Show process on the container
docker exec looper ps -ef
docker rm -f looper

docker build -t loop .
docker run -d --name looper loop
docker exec looper ps -ef
docker stop looper
docker ps -l

# Ping
docker build -t ping .
docker run --name pinger ping
docker rm pinger

docker build -t ping .
docker run --name pinger ping
docker rm pinger
# Override command
docker run --name pinger ping google.com

# Multi-stage
docker build -t prodapp -f build/production.Dockerfile .
docker run -d --name prod prodapp
docker ps
docker exec -it prod bash

# Docker-in-Docker
docker run -it --rm -v /var/run/docker.sock:/var/run/docker.sock docker:19.03.12
> docker ps
> docker run -d --name app prodapp
> docker ps

# CodelyTV
docker build -t "cmd-vs-entrypoint" .
docker run --rm -it cmd-vs-entrypoint /etc/os-release
docker run --rm -it --entrypoint /bin/bash cmd-vs-entrypoint

docker run --rm -p 8000:80 "environment-vars"
